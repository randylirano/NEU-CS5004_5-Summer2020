public class RightTri extends Tri {

    private double base, height, side3;
    private int color;

    public RightTri(double base, double height, int color) {
        super(base, height, 0, color);
        this.side3 = Math.sqrt(base * base + height * height);
        this.base = base;
        this.height = height;
        }
public abstract class Shape implements Comparable<Shape>, ColorfulThing {

    public abstract double area();
    public abstract double perimeter();
    public abstract int getColor();

    // compare the area of this shape to another one
    // return true iff this one is larger

    public int compareTo(Shape o) {
        if (this.area() < o.area()) {
            return -1;
        } else if (this.area() > o.area()) {
            return 1;
        } else {
            // equal
            return 0;
        }
    }
